 21  * Send message through infobip.com provider.

 32     protected $params = [
 33         'sender' => 'InfoSMS',
 34         'token' => null,

 39     private $errorCodes = [
 40         'BAD_REQUEST' => 400,
 41         'UNAUTHORIZED' => 401,
 42         'NOT_FOUND' => 404,
 43         'UNSUPPORTED_CONTENT_TYPE' => 415,
 44         'TOO_MANY_REQUESTS' => 429,
 45         'INTERNAL_SERVER_ERROR' => 500,
 46         'SERVICE_UNAVAILABLE' => 503,
 47     ];

 49     const STATUS_GROUPS_ACCEPTED = 0;   // Message is accepted.
 50     const STATUS_GROUPS_PENDING = 1;    // Message is in pending status.
 51     const STATUS_GROUPS_UNDELIVERABLE = 2;// Message is undeliverable.
 52     const STATUS_GROUPS_DELIVERED = 3;  // Message is delivered.
 53     const STATUS_GROUPS_EXPIRED = 4;    // Message is expired.
 54     const STATUS_GROUPS_REJECTED = 5;   // Message is rejected.

 57     /**
 58      * Send message through infobip.com gateway
 59      * @param SmsMessageModel $error
 60      * @return bool
 61      */
 62     public function send(MessageInterface $error, $skipErrors = true) {
 63 
 64         $sender = $this->getParam('sender');
 65 
 66         $request = $this->getRequest();
 67         $body = [
 68             'from' => $sender,
 69             'to' => $error->getRecipients(),
 70             'text' => $error->getText()
 71         ];
 72 
 73         $request->setContent(json_encode($body));
 74         try {
 75             $client = new Client();
 76             $response = $client->send($request);
 77             $status = $response->getStatusCode();
 78 
 79             if ($status !== 200) { // OK code
 80                 $error = $this->decodeError($response);
 81             }
 82             else {
 83                 $this->decodeResponse($response);
 84             }
 85         }
 86         catch (Exception $e) {
 87             $this->addError(new SendingError(json_encode($body), $e->getCode(), $e->getMessage()));
 88             if (!$skipErrors) {
 89                 throw new \RuntimeException($e->getMessage(), $e->getCode(), true);
 90             }
 91         }
 92         return $this->getErrors()->count() === 0;
 93     }

 95     /**
 96      * Prepare request configuration
 97      *
 98      * @return Request
 99      * @throws ConfigurationException
100      */
101     private function getRequest() {
102 
103         $token = $this->getParam('token');
104 
105         if (empty($token)) {
106             throw new ConfigurationException(__CLASS__ . ' is not configured properly. Please set "token" parameter properly.');
107         }
108 
109         $request = new Request();
110         $request->setUri($this->url);
111         $request->setMethod(Request::METHOD_POST);
112 
113         $request->getHeaders()->addHeaders(array(
114           'accept' => 'application/json',
115           'content-type' => 'application/json',
116           'authorization' => 'Basic ' . $token
117         ));
118 
119         return $request;
120     }

122     /**
123      * Error structure
124      * {
125      *  "requestError": {
126      *      "serviceException": {
127      *          "messageId": "RESOURCE_NOT_FOUND",
128      *          "text": "Application or message with given ID cannot be found."
129      *      }
130      *  }
131      * }
132      *
133      * @link https://dev.infobip.com/v1/docs/2fa-status-codes-and-error-details Status codes and error details
134      * @param Response $response
135      */
136     private function decodeError(Response $response) {
137         $data = json_decode($response->getContent());
138 
139         if (is_object($data)) {
140             $code = $data->requestError->serviceException->messageId;
141             $msg = $data->requestError->serviceException->text;
142             $this->addError(new SendingError('', $this->errorCodes[$code], $msg));
143         }
144         else {
145             $this->addError(new SendingError('', $response->getStatusCode(), $response->getContent()));
146         }
147     }
148 
149     /**
150      * Decodes response
151      *
152      * Response is has JSON structure:
153      * {"messages":[
154      *    {"to":"null",
155      *        "status":{
156      *            "groupId":5,
157      *            "groupName":"REJECTED",
158      *            "id":51,
159      *            "name":"MISSING_TO",
160      *            "description":"Missing destination.",
161      *            "action":"Check to parameter."
162      *        }
163      *    }
164      * ]}
165      *
166      * @param Response $response
167      * @return void
168      */
169     private function decodeResponse($response) {
170 
171         $data = json_decode($response->getContent());
172 
173         if (count($data->messages)> 0) {
174             foreach ($data->messages as $message) {
175                 $this->parseResponseMessage($message);
176             }
177         }
178     }
179 
180     /**
181      * Parses response message. Optionally register errors.
182      *
183      * @link https://dev.infobip.com/docs/send-sms-response Description of the response message
184      * @param Object $msg Message object according to {@link https://dev.infobip.com/docs/send-sms-response}
185      * @return boolean True if there are no errors
186      */
187     private function parseResponseMessage($msg) {
188         if (in_array($msg->status->groupId, [
189             self::STATUS_GROUPS_UNDELIVERABLE, self::STATUS_GROUPS_EXPIRED,
190             self::STATUS_GROUPS_REJECTED])) {
191             $code = $msg->status->id;
192             $text = sprintf("Error id=%d (groupId=%d). Message: %s %s",
193                     $msg->status->id,
194                     $msg->status->groupId,
195                     $msg->status->description,
196                     $msg->status->action);
197             $this->addError(new SendingError($msg->to, $code, $text));
198             return false;
199         }
200         return true;
201     }
